
def download_pdf(format_dated, format_date, order_list, row, case_number, case_board_date, case_type, search_stamp_no, year, casetype_dict, current_time, date_pattern_fmts, date_replace_pattern, seq_no, order_file_name):
    base_url='https://bombayhighcourt.nic.in/'
    url='generatenewauth.php?bhcpar='
    query='path=./writereaddata/data/civil/2025/&fname={stamp_no}{case_type}{case_number}{year}_{seq}.pdf&smflag=N&rjuddate=&uploaddt=&spassphrase={current_time}&ncitation=&smcitation=&digcertflg=&interface='
    count = 0
    while True:
        count = count + 1
        query_fmt = None
        if case_type in casetype_dict.keys():
            query_fmt = query.format(case_type = casetype_dict[case_type], 
                                        case_number = case_number, 
                                        year = year, seq = str(seq_no), 
                                        current_time = current_time,
                                        stamp_no = 'F' if search_stamp_no else '')
        else:
            print('No config found for Case type ' + case_type)
            break
            
        seq_no = seq_no + 1
        query_utf_8 = query_fmt.encode('utf-8')
        encoded_query = base64.b64encode(query_utf_8)
        query_str = encoded_query.decode('utf-8')
        full_url = base_url + url + query_str

        response = requests.get(full_url)

        if 'application/pdf' == response.headers['Content-Type']:
            writer = PdfWriter()
            pdf_data = PdfReader(BytesIO(response.content))
            pages = len(pdf_data.pages)
            order_found = False
            for i in range(pages):
                page = pdf_data.pages[i]
                writer.add_page(page)
                text = page.extract_text()
                lines = text.splitlines()
                for line in lines:
                    for pattern in order_list:
                        matches = re.match(pattern, line)
                        if matches:
                            if pattern == format_date or pattern == format_dated:
                                date_formatted = matches.group(1)
                                head, sep, tail = text.partition('2024')
                                date_formatted = date_formatted.replace(tail, '')
                                for sub in date_replace_pattern:
                                    date_formatted = date_formatted.replace(sub, '')
                                date_formatted = date_formatted.replace(' ', '')
                                
                                if 'AUGUST' in date_formatted.upper():
                                    date_formatted = date_formatted
                                elif 'AUGU' in date_formatted.upper():
                                    date_formatted = date_formatted.upper().replace('AUGU', 'AUGUST')
                                
                                try:
                                    order_date = None
                                    for format in date_pattern_fmts:
                                        order_date = datetime.strptime(date_formatted, format)
                                        
                                        if order_date == case_board_date:
                                            order_found = True
                                        break
                                except ValueError as e:
                                    print('Date format invalid: ' + date_formatted)
                                    pass
            if order_found:
                print('Order found for : ' + order_file_name)
                writer.write('orders' + '/' + order_file_name)
                break
        if count > 20:
            print('No order found for - ' + row['case_no'])
            break
